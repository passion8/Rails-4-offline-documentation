def _attribute(attr_name)
  ActiveSupport::Deprecation.warn(
    "You have called '_#{attr_name}'. This is deprecated. Please use "            "either '#{attr_name}' or read_attribute('#{attr_name}')."
  )
  read_attribute(attr_name)
end
