def destroy
  clear
  options = @env[Rack::Session::Abstract::ENV_SESSION_OPTIONS_KEY] if @env
  options ||= {}
  @by.send(:destroy_session, @env, options[:id], options) if @by
  options[:id] = nil
  @loaded = false
end
