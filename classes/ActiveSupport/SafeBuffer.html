def [](*args)
  return super if args.size < 2

  if html_safe?
    new_safe_buffer = super
    new_safe_buffer.instance_eval { @html_safe = true }
    new_safe_buffer
  else
    to_str[*args]
  end
end

def concat(value)
  if !html_safe? || value.html_safe?
    super(value)
  else
    super(ERB::Util.h(value))
  end
end
